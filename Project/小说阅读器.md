### 一、项目概述

**项目名称**：SimpleReader  
**定位**：极简、干净、高性能的本地电子书阅读器（支持 TXT + EPUB）  
**核心特色**：
- 纯本地阅读，无广告、无云同步、无隐私弹窗
- 完整的语音听书（多音色 + 句子高亮 + 自动记忆位置 + 后台播放）
- Room 数据库持久化书架与阅读进度
- 夜间模式、字体调节、简洁 UI

**已实现全部功能**：
- 书架（Grid 展示）
- 从任意位置导入 TXT / EPUB 文件
- 完整阅读器（TXT + EPUB）
- 字体大小调节
- 日间 / 夜间模式
- **语音听书**（多音色选择 + SpannableString 高亮当前句子 + 记忆听书位置）
- **后台播放 Service**（前台服务 + 通知栏）
- **Room 数据库** 持久化书架、阅读进度、听书位置

```kotlin
package com.simplereader.database

import android.content.Context
import androidx.room.Database
import androidx.room.Room
import androidx.room.RoomDatabase
import com.simplereader.model.BookEntity

@Database(entities = [BookEntity::class], version = 1)
abstract class AppDatabase : RoomDatabase() {
    abstract fun bookDao(): BookDao

    companion object {
        @Volatile
        private var INSTANCE: AppDatabase? = null

        fun getDatabase(context: Context): AppDatabase {
            return INSTANCE ?: synchronized(this) {
                Room.databaseBuilder(context, AppDatabase::class.java, "simplereader_db").build()
                    .also { INSTANCE = it }
            }
        }
    }
}
```

```kotlin
package com.simplereader.database

import androidx.room.Dao
import androidx.room.Insert
import androidx.room.OnConflictStrategy
import androidx.room.Query
import com.simplereader.model.BookEntity

@Dao
interface BookDao {
    @Query("SELECT * FROM books ORDER BY lastReadTime DESC")
    suspend fun getAllBooks(): List<BookEntity>

    @Insert(onConflict = OnConflictStrategy.REPLACE)
    suspend fun insertBook(book: BookEntity)

    @Query("UPDATE books SET lastReadPosition = :position, lastTtsPosition = :ttsPos, lastReadTime = :time WHERE id = :id")
    suspend fun updateProgress(id: String, position: Float, ttsPos: Int, time: Long = System.currentTimeMillis())
```

#### 6. **核心阅读器 + 全部高级功能** `reader/ReaderActivity.kt`

```kotlin
package com.simplereader.reader

import android.content.Intent
import android.graphics.Color
import android.os.Bundle
import android.speech.tts.TextToSpeech
import android.speech.tts.UtteranceProgressListener
import android.text.Spannable
import android.text.SpannableString
import android.text.style.BackgroundColorSpan
import android.widget.Toast
import androidx.appcompat.app.AlertDialog
import androidx.appcompat.app.AppCompatActivity
import androidx.lifecycle.lifecycleScope
import com.simplereader.database.AppDatabase
import com.simplereader.databinding.ActivityReaderBinding
import com.simplereader.model.Book
import com.simplereader.model.BookEntity
import com.simplereader.service.TtsService
import kotlinx.coroutines.launch
import org.folio.reader.FolioReader
import java.io.File
import java.util.Locale

class ReaderActivity : AppCompatActivity(), TextToSpeech.OnInitListener {

    private lateinit var binding: ActivityReaderBinding
    private var currentBook: Book? = null
    private var bookEntity: BookEntity? = null

    private lateinit var tts: TextToSpeech
    private var isTtsInitialized = false
    private var isSpeaking = false

    private val sentences = mutableListOf<String>()
    private var currentSentenceIndex = 0

    private val db by lazy { AppDatabase.getDatabase(this) }

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        binding = ActivityReaderBinding.inflate(layoutInflater)
        setContentView(binding.root)

        currentBook = intent.getSerializableExtra("book") as? Book
        currentBook?.let { loadBook(it) }

        tts = TextToSpeech(this, this)

        setupListeners()
    }

    private fun loadBook(book: Book) {
        title = book.title

        lifecycleScope.launch {
            bookEntity = db.bookDao().getAllBooks().find { it.id == book.id }
                ?: BookEntity(id = book.id, title = book.title, author = book.author, path = book.path, type = book.type.name)
            db.bookDao().insertBook(bookEntity!!)
        }

        if (book.type == BookType.EPUB) {
            FolioReader.get().openBook(book.path)
        } else {
            val content = File(book.path).readText()
            binding.tvContent.text = content
            sentences.addAll(content.split(Regex("[。！？]")).map { it.trim() }.filter { it.isNotEmpty() })
        }

        currentSentenceIndex = bookEntity?.lastTtsPosition ?: 0
    }

    private fun setupListeners() {
        binding.btnFontSmall.setOnClickListener { adjustFontSize(-2f) }
        binding.btnFontLarge.setOnClickListener { adjustFontSize(2f) }
        binding.btnNightMode.setOnClickListener { toggleNightMode() }

        binding.fabTts.setOnClickListener {
            if (isSpeaking) stopTts() else startTts()
        }

        binding.fabTts.setOnLongClickListener {
            showVoiceSelectionDialog()
            true
        }
    }

    override fun onInit(status: Int) {
        if (status == TextToSpeech.SUCCESS) {
            isTtsInitialized = true
            tts.language = Locale.CHINA
            setupTtsListener()
        }
    }

    private fun setupTtsListener() {
        tts.setOnUtteranceProgressListener(object : UtteranceProgressListener() {
            override fun onStart(utteranceId: String?) { runOnUiThread { highlightCurrentSentence() } }
            override fun onDone(utteranceId: String?) {
                currentSentenceIndex++
                if (currentSentenceIndex < sentences.size) speakCurrentSentence()
                else isSpeaking = false
            }
            override fun onError(utteranceId: String?) {}
        })
    }

    private fun startTts() {
        if (!isTtsInitialized || sentences.isEmpty()) {
            Toast.makeText(this, "TTS 未就绪", Toast.LENGTH_SHORT).show()
            return
        }
        isSpeaking = true
        speakCurrentSentence()

        // 启动后台服务
        startService(Intent(this, TtsService::class.java).apply {
            putExtra("bookId", currentBook?.id)
        })
    }

    private fun speakCurrentSentence() {
        if (currentSentenceIndex >= sentences.size) return
        val text = sentences[currentSentenceIndex]
        tts.speak(text, TextToSpeech.QUEUE_FLUSH, null, "sentence_$currentSentenceIndex")
    }

    private fun stopTts() {
        tts.stop()
        isSpeaking = false
        saveProgress()
    }

    private fun highlightCurrentSentence() {
        if (sentences.isEmpty()) return
        val fullText = binding.tvContent.text.toString()
        val sentence = sentences.getOrNull(currentSentenceIndex) ?: return

        val spannable = SpannableString(fullText)
        val start = fullText.indexOf(sentence)
        if (start != -1) {
            spannable.setSpan(BackgroundColorSpan(Color.YELLOW), start, start + sentence.length, Spannable.SPAN_EXCLUSIVE_EXCLUSIVE)
            binding.tvContent.text = spannable
        }
    }

    private fun showVoiceSelectionDialog() {
        val voices = tts.voices?.filter { it.locale.language == "zh" } ?: emptyList()
        AlertDialog.Builder(this)
            .setTitle("选择音色")
            .setItems(voices.map { it.name }.toTypedArray()) { _, which ->
                tts.voice = voices[which]
                Toast.makeText(this, "已切换：${voices[which].name}", Toast.LENGTH_SHORT).show()
            }.show()
    }

    private fun adjustFontSize(delta: Float) {
        val current = binding.tvContent.textSize / resources.displayMetrics.scaledDensity
        binding.tvContent.textSize = current + delta
    }

    private fun toggleNightMode() {
        val nightMode = if (resources.configuration.uiMode and android.content.res.Configuration.UI_MODE_NIGHT_MASK ==
            android.content.res.Configuration.UI_MODE_NIGHT_YES) androidx.appcompat.app.AppCompatDelegate.MODE_NIGHT_NO
        else androidx.appcompat.app.AppCompatDelegate.MODE_NIGHT_YES
        androidx.appcompat.app.AppCompatDelegate.setDefaultNightMode(nightMode)
        recreate()
    }

    private fun saveProgress() {
        currentBook?.let { book ->
            lifecycleScope.launch {
                db.bookDao().updateProgress(book.id, 0f, currentSentenceIndex)
            }
        }
    }

    override fun onPause() {
        super.onPause()
        if (isSpeaking) saveProgress()
    }

    override fun onDestroy() {
        tts.shutdown()
        super.onDestroy()
    }
}
```

#### 7. 后台播放服务 `service/TtsService.kt`

```kotlin
package com.simplereader.service

import android.app.Notification
import android.app.NotificationChannel
import android.app.NotificationManager
import android.app.Service
import android.content.Intent
import android.os.Build
import android.os.IBinder
import androidx.core.app.NotificationCompat
import com.simplereader.R

class TtsService : Service() {

    override fun onCreate() {
        super.onCreate()
        createNotificationChannel()
        startForeground(1, createNotification())
    }

    override fun onStartCommand(intent: Intent?, flags: Int, startId: Int): Int {
        return START_STICKY
    }

    private fun createNotification(): Notification {
        return NotificationCompat.Builder(this, "tts_channel")
            .setContentTitle("正在听书")
            .setContentText("点击返回阅读界面")
            .setSmallIcon(R.drawable.ic_book)
            .setPriority(NotificationCompat.PRIORITY_LOW)
            .build()
    }

    private fun createNotificationChannel() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            val channel = NotificationChannel("tts_channel", "听书服务", NotificationManager.IMPORTANCE_LOW)
            getSystemService(NotificationManager::class.java).createNotificationChannel(channel)
        }
    }

    override fun onBind(intent: Intent?): IBinder? = null
    override fun onDestroy() { super.onDestroy() }
}
```
